    %% Entities (Tables)
    users {
        BIGINT uid PK "用户ID"
        VARCHAR(20) phone "用户手机号"
        VARCHAR(255) password "用户密码"
        BOOLEAN is_first_login "是否第一次登录"
        BOOLEAN status "用户状态"
        tinyint is_vip "是否是VIP用户"
        datetime vip_expire_time "VIP过期时间"
        tinyint vip_level "VIP等级"
        VARCHAR(50) vip_source "VIP来源"
        DATETIME created_at "用户创建时间"
        DATETIME updated_at "用户更新时间"
    }

    movie {
        BIGINT mid PK "电影ID"
        VARCHAR(50) did "豆瓣ID"
        VARCHAR(255) title "电影名称"
        DOUBLE rate "电影评分"
        VARCHAR(255) cover "电影海报URL"
        VARCHAR(255) directors "电影导演"
        VARCHAR(255) writer "电影编剧"
        VARCHAR(255) casts "电影演员阵容"
        VARCHAR(255) types "电影类型"
        VARCHAR(100) country "电影上映国家/地区"
        VARCHAR(100) language "电影语言"
        VARCHAR(50) release_date "电影上映日期"
        DATETIME release_newest_date "电影上映最新日期"
        VARCHAR(50) movie_time "电影时长"
        VARCHAR(255) aka "电影又名"
        TEXT summary "电影简介"
        VARCHAR(255) trailer_url "电影预告片网上链接"
        VARCHAR(255) trailer_minio_url "电影预告片本地路径"
    }

    viewing_history {
        BIGINT id PK "主键ID"
        INT user_id FK "用户ID"
        INT movie_id FK "电影ID"
        DATETIME last_watch_time "最后观看时间"
        INT current_time "当前播放位置（秒）"
        INT total_time "影片总时长（秒）"
        INT progress "观看进度百分比（0~100）"
        INT watch_count "观看次数"
        DATETIME created_at "首次观看时间"
        DATETIME updated_at "最近更新时间"
    }

    user_behavior_log {
        BIGINT id PK "主键ID"
        BIGINT uid FK "用户ID"
        BIGINT mid FK "电影ID"
        VARCHAR(50) behavior_type "行为类型（VIEW、SEARCH、RATE等）"
        VARCHAR(255) keyword "搜索关键词"
        BIGINT duration "停留时长（毫秒）"
        DOUBLE rating "评分值"
        TEXT comment "标签内容"
        VARCHAR(500) danmaku "弹幕内容"
        BIGINT result_position "搜索结果位置"
        VARCHAR(50) ip_address "用户IP地址"
        DATETIME create_time "创建时间"
    }

    danmaku {
        BIGINT id PK "主键ID"
        INT movie_id FK "电影ID"
        BIGINT user_id FK "用户ID"
        VARCHAR(500) text "弹幕文本"
        DOUBLE time "弹幕时间点"
        VARCHAR(20) color "弹幕颜色"
        VARCHAR(20) type "弹幕类型（SCROLL、TOP、BOTTOM）"
        DATETIME created_at "创建时间"
        DATETIME updated_at "更新时间"
    }

    user_movie_comment {
        BIGINT id PK "主键ID"
        INT mid FK "电影ID"
        INT uid FK "用户ID"
        TEXT content "评论内容"
        DOUBLE sentiment_score "情感得分（0~1）"
        DATETIME create_time "创建时间"
        DATETIME update_time "更新时间"
    }

    user_movie_rate {
        BIGINT id PK "主键ID"
        INT mid FK "电影ID"
        INT uid FK "用户ID"
        DOUBLE score "评分值（1~10）"
        DATETIME create_time "创建时间"
    }

    movie_types {
        INT id PK "类型ID"
        VARCHAR(50) type "类型名称"
    }

    banner {
        INT mid PK "轮播图ID"
        VARCHAR(255) title "标题"
        VARCHAR(255) title_img "标题图片URL"
        VARCHAR(255) img "轮播图URL"
        VARCHAR(50) category "类别"
        TEXT description "描述"
        VARCHAR(255) video_url "视频URL"
        INT sort "排序"
        BOOLEAN status "状态（0-禁用，1-启用）"
        DATETIME create_time "创建时间"
    }

    %% Relationships
    users ||--|{ viewing_history : "观看"
    movie ||--|{ viewing_history : "被观看"

    users ||--|{ user_behavior_log : "产生行为"
    movie ||--o{ user_behavior_log : "行为对象" %% mid in log is NULLable

    users ||--|{ danmaku : "发送弹幕"
    movie ||--|{ danmaku : "拥有弹幕"

    users ||--|{ user_movie_comment : "发表评论"
    movie ||--|{ user_movie_comment : "被评论"

    users ||--|{ user_movie_rate : "进行评分"
    movie ||--|{ user_movie_rate : "被评分"
